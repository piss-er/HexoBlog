};

class Solution {
public:
    string reverseStr(string s, int k) {
        int count = 0;
        int len = s.size();
        int left,right;
        int flag = 0;
        while(flag<len){
            left = flag;
            while(count<2*k && flag<len){
                count++;
                flag++;
            }
            right = left+min(count,k)-1;
            while(left<right){
                swap(s[left++],s[right--]);
            }
            count = 0;
        }
        return s;
    }
};
